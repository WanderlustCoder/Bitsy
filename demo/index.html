Characters

Professional-quality pixel art characters with CharacterBuilder API, presets, equipment, and species variants.

from characters import CharacterBuilder, presets

# Using presets
hero = presets.hero(hair_color='brown', width=48)

# Using fluent builder with equipment
knight = (CharacterBuilder(width=48, height=48)
    .head('square').body('muscular')
    .hair('short', color='blonde')
    .eyes('round', color='blue')
    .equip_set('knight')
    .build())

HD Features

Professional-quality pixel art with anti-aliasing, selective outline (selout), and multi-level shading.

from core.style import PROFESSIONAL_HD
from quality.selout import apply_selout

# HD style with 6-level shading
char = CharacterBuilder(width=64).style(PROFESSIONAL_HD).build()
sprite = apply_selout(char.render()) # Color-aware outlines
